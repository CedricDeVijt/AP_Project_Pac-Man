168 // Get the current position and size of the entity

173 const double toleranceY = sizeY * tolerance;

175 // Initialize possible directions with all directions initially
176 std::vector <Direction> possibleDirections = {LEFT, RIGHT, UP, DOWN};
177
178 // Check if entity collides with the wall
179 for (auto &wall : walls) {
180 // Get wall's position and size
181 double wallX, wallY, wallSizeX, wallSizeY;
182 std::tie(wallX, wallY, wallSizeX, wallSizeY) = wall->getPosition();
183
184 // Check UP DOWN, reduce width of box
185 // Check for collision along the X-axis
186 bool collisionX = x + toleranceX <= wallX + wallSizeX && x + sizeX - toleranceX >= wallX;
187 bool collisionY = y - toleranceY <= wallY + wallSizeY && y + sizeY + toleranceY >= wallY;
188 // If there is a collision along both axes, PacMan cannot move in that direction
189 if (collisionX && collisionY) {
190 if (y <= wallY) {
191 possibleDirections.erase(std::remove(possibleDirections.begin(), possibleDirections.end(), DOWN),
192 possibleDirections.end());
193 } else if (y + sizeY >= wallY + wallSizeY) {
194 possibleDirections.erase(std::remove(possibleDirections.begin(), possibleDirections.end(), UP),
195 possibleDirections.end());

197 }

199 // Check LEFT & RIGHT, reduce Height of box
200 // Check for collision along the X-axis and Y-axis
201 collisionX = x - toleranceX <= wallX + wallSizeX && x + sizeX + toleranceX >= wallX;
202 collisionY = y + toleranceY <= wallY + wallSizeY && y + sizeY - toleranceY >= wallY;
203 // If there is a collision along both axes, PacMan cannot move in that direction
204 if (collisionX && collisionY) {
