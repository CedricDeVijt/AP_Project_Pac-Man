188bool Ghost::atCornerOrIntersection(const std::vector<Direction>& directions) {
189 // Check if "up" or "down" is present
190 const bool hasUpDown = std::find(directions.begin(), directions.end(), Direction::UP) != directions.end() ||
191 std::find(directions.begin(), directions.end(), Direction::DOWN) != directions.end();
192
193 // Check if "left" or "right" is present
194 const bool hasLeftRight = std::find(directions.begin(), directions.end(), Direction::LEFT) != directions.end() ||
195 std::find(directions.begin(), directions.end(), Direction::RIGHT) != directions.end();
196
197 // Return true if one of "up" or "down" is present AND one of "left" or "right" is present
198 return hasUpDown && hasLeftRight;
